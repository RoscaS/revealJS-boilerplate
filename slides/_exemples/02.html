      template<class T>
      class Babysitter {
         T *ptr;
      public:
         Babysitter(T *ptr = nullptr) : ptr(ptr) {}
         ~Babysitter() { delete ptr; }

         Babysitter(Babysitter& other) {
             ptr = other.ptr;
             other.ptr = nullptr;
         }

         bool isNull() const { return ptr == nullptr; }
         T &operator*() const { return *ptr; }
         T *operator->() const { return ptr; }
      };
